    # 一意なARIMA(p,q)モデルをすべて適合させる関数
def optimize_ARIMA(endog: Union[pd.Series, list], orders: list, d: int) -> pd.DataFrame:
    """
    endog: 時系列データ
    orders: ARIMA(p, d, q)のpとqの値の組み合わせ
    d: 和分次数
    """
    res = []
    for order in tqdm(orders):
        try:
            model = SARIMAX(
                endog, 
                order=(order[0], d, order[1]),
                simple_differencing=False
            ).fit(disp=False)
            res.append([order, model.aic])
        except:
            continue
    df_res = (
        pd.DataFrame(res, columns=['(p,q)', 'AIC'])
        .sort_values(by='AIC', ascending=True)
        .reset_index(drop=True)
    )
    return df_res
    # 一意なARIMA(p,q)モデルをすべて適合させる関数 def optimize_ARIMA(endog: Union[pd.Series, list], orders: list, d: int) -> pd.DataFrame: """ endog: 時系列データ orders: ARIMA(p, d, q)のpとqの値の組み合わせ d: 和分次数 """ res = [] for order in tqdm(orders): try: model = SARIMAX( endog, order=(order[0], d, order[1]), simple_differencing=False ).fit(disp=False) res.append([order, model.aic]) except: continue df_res = ( pd.DataFrame(res, columns=['(p,q)', 'AIC']) .sort_values(by='AIC', ascending=True) .reset_index(drop=True) ) return df_res
